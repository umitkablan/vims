MRU Window Toggle
-----------------

function! s:Mru_Window_Toggle()
    let wnum = bufwinnr('__MRU_Files__')

    " Open or jump to the MRU window
    MRU

    if wnum != -1
        " MRU window was already open. Close it
        close
    endif
endfunction

nnoremap <silent> <F7> :call <SID>Mru_Window_Toggle()<CR>
